  geometry_msgs
  nav_msgs
  sensor_msgs
  roscpp
  rospy
  std_msgs
  laser_assembler
  tf)

#find_package(Eigen3 REQUIRED)

  include
	${catkin_INCLUDE_DIRS}

	${PCL_INCLUDE_DIRS})

  CATKIN_DEPENDS geometry_msgs nav_msgs roscpp rospy std_msgs
